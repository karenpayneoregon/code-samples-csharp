public class Form1
{
	private void RemoveFoldersButton_Click(object sender, EventArgs e)
	{
		Operations.RecursiveDelete(new DirectoryInfo("C:\\OED\\DataGridViewCheckBoxOneRow"));
	}

	private void Form1_Load(object sender, EventArgs e)
	{
		Operations.OnExceptionEvent += ExceptionEvent;
		Operations.OnDeleteEvent += DeletingEvent;
	}

	private void DeletingEvent(string item)
	{
		Console.WriteLine(item);
	}

	private void ExceptionEvent(Exception exception)
	{
		Console.WriteLine(exception.Message);
	}

	public Form1()
	{
		SubscribeToEvents();
	}
}
